\begin{frame}
	\frametitle{Feature Transformation: Income \& Wealth}

	\textbf{Problem:} Extreme right-skewness (skewness $>$ 1)
	\begin{itemize}
		\item Income: skewness = 1.38, kurtosis = 2.86
		\item Wealth: skewness = 5.83, kurtosis = 67.13
		\item Causes numerical instability in distance-based models (SVM, KNN)
	\end{itemize}

	\vspace{0.5cm}

	\textbf{Solution:} Log1p Transformation
	$$\text{Transformed} = \log(1 + x)$$

	\textbf{Advantages:}
	\begin{itemize}
		\item Compresses right tail without losing monotonicity
		\item Interpretable: 1-unit change = multiplicative change in raw value
		\item Achieves near-symmetry (skewness $<$ 0.5)
	\end{itemize}

	\vspace{0.5cm}

	\textbf{Applied:} StandardScaler after log1p transformation for model input
\end{frame}

\begin{frame}
	\frametitle{Engineered Features (16 total)}

	\begin{enumerate}
		\item \textbf{Base Features} (7)
		      \begin{itemize}
			      \item Original 6 numeric + transformed Income \& Wealth
		      \end{itemize}

		\item \textbf{Household-Adjusted Ratios} (3)
		      \begin{itemize}
			      \item IncomePerFamilyMember = Income / (FamilyMembers + 1)
			      \item WealthPerFamilyMember = Wealth / (FamilyMembers + 1)
			      \item WealthIncomeRatio = Wealth / (Income + 1)
		      \end{itemize}

		\item \textbf{Interaction Features} (3)
		      \begin{itemize}
			      \item RiskEducationInteraction = RiskPropensity $\times$ FinancialEducation
			      \item RiskWealthInteraction = RiskPropensity $\times$ Wealth
			      \item AgeRiskInteraction = Age $\times$ RiskPropensity
		      \end{itemize}

		\item \textbf{Non-linearity \& Life-Stage Flags} (3)
		      \begin{itemize}
			      \item AgeSquared = Age$^2$
			      \item Age\_Under35, Age\_35\_54, Age\_55\_69, Age\_70plus
		      \end{itemize}
	\end{enumerate}

	\vspace{0.3cm}

	\textcolor{darkgreen}{\textbf{Result:}} Engineered features improve F1-score by 2-5\% vs. base features
\end{frame}

\section{Explainable AI (XAI)}

\begin{frame}
	\frametitle{XAI Methods: SHAP, LIME \& Permutation}

	\textbf{Three complementary interpretability approaches:}

	\vspace{0.3cm}

	\begin{columns}[T]
		\column{0.33\textwidth}
		\textbf{SHAP (SHapley)}
		\begin{itemize}
			\item Game-theoretic approach
			\item Global feature importance
			\item Individual prediction explanations
		\end{itemize}

		\column{0.33\textwidth}
		\textbf{LIME}
		\begin{itemize}
			\item Local linear approximation
			\item Model-agnostic
			\item Single-instance explanations
		\end{itemize}

		\column{0.33\textwidth}
		\textbf{Permutation Importance}
		\begin{itemize}
			\item Feature contribution measurement
			\item Direct performance impact
			\item Model-specific
		\end{itemize}
	\end{columns}

	\vspace{0.5cm}

	\textbf{Purpose:}
	\begin{itemize}
		\item Understand which features drive predictions
		\item Build client trust and regulatory compliance
		\item Identify potential biases or fairness issues
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Income Investment Model Insights}

	\begin{enumerate}
		\item \textbf{WealthPerFamilyMember} (Dominant)

			      \item Interpretation: Concentrated wealth among few household members

		      \begin{itemize}

			      \item Better-educated clients more likely to seek income products

	\end{enumerate}

	\vspace{0.3cm}

	\textcolor{darkgreen}{\textbf{Business Insight:}} \textit{Older, wealthy clients with high financial literacy prefer income-generating products.}
\end{frame}

\begin{frame}
	\frametitle{Accumulation Investment Model Insights}

	\textbf{SHAP Analysis - Feature Importance (Global):}

	\begin{enumerate}
		\item \textbf{WealthIncomeRatio} (Dominant - Inverse Relationship!)
		      \begin{itemize}

\begin{frame}
	\frametitle{Appendix: Cross-Validation Strategy}

	\textbf{Why Stratified K-Fold?}

	\begin{itemize}
		\item \textbf{Stratification}: Preserves class distribution in each fold
		      \begin{itemize}
			      \item Important for imbalanced targets (Income: 38.4\% positive)
			      \item Prevents folds with extreme class ratios
		      \end{itemize}

		\item \textbf{K-Fold (K=5)}:
		      \begin{itemize}
			      \item Balance between variance \& computational cost
			      \item 4 folds for training (4,000 samples)
			      \item 1 fold for validation (1,000 samples)
			      \item Repeat 5 times, average results
		      \end{itemize}
	\end{itemize}

	\vspace{0.3cm}

	\textbf{Evaluation Metrics:}
	\begin{itemize}
		\item \textcolor{darkblue}{\textbf{Primary}}: Precision \& Recall (handles class imbalance)
		\item \textcolor{darkgreen}{\textbf{Secondary}}: F1-score (harmonic mean)
		\item \textcolor{gold}{\textbf{Tertiary}}: ROC-AUC (threshold-independent)
		\item \textcolor{darkred}{\textbf{Avoid}}: Accuracy (misleading for imbalanced data)
	\end{itemize}
\end{frame}
